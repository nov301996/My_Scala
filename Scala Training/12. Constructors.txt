
class Employee (var EmpId:Int,var EmpName:String,var Empsal:Float) {
  var EmployeeId:Int=EmpId
  var EmployeeName:String=EmpName
  var EmployeeSal:Float=Empsal

  def hikeSalary(percent:Float): Unit =
  {
    EmployeeSal=EmployeeSal+(percent*EmployeeSal)
    println(s"EmpID: ${EmployeeId}")
    println(s"EmpName: ${EmployeeName}")
    println(s"EmpSal: ${EmployeeSal}")
  }

}

object Demo {
  def main(args: Array[String]): Unit =
  {
    val Empl =new Employee(1,"Kartik",50000)
    Empl.hikeSalary(0.20f)
  }
}
